CREATE TABLE Delicacy (
  "id" Int GENERATED ALWAYS AS IDENTITY,
  "Name" VarChar(60) UNIQUE NOT NULL,
  "Quantity" Int NOT NULL,
  "Price" float(3) NOT NULL,
  "Weight" float(3) NOT NULL,
  "Nutritional_energy" float(3) NOT NULL,
  "Veg" Boolean,
  "Special" Boolean,
  primary key ("id")
);

CREATE TABLE Admin (
  "id"  Int GENERATED ALWAYS AS IDENTITY,
  "Name" VarChar(40),
  "UserName" VarChar(50),
  "Password" VarChar(20),
  primary key ("id")
  
  
);

CREATE TABLE Delicacy_Scheduled (
  "id"  Int GENERATED ALWAYS AS IDENTITY,
  "delicacy_id" int, 
  "date" Date,
  FOREIGN KEY("delicacy_id")
  REFERENCES Delicacy("id")
);
